** PSDJAVA-566. Manipular tipo de gradiente Noise em GdflResource**

{{< highlight java >}}

    String arquivoFonte = "src/main/resources/Gradient-Fill.psd";
    String arquivoDestino = "src/main/resources/Gradient-Fill-out.psd";

    try (PsdImage imagem = (PsdImage) Image.load(arquivoFonte, new PsdLoadOptions())) {
        Layer camada = imagem.getLayers()[1];

        for (LayerResource recurso : camada.getResources()) {
            GdFlResource recursoGdFl = (GdFlResource) recurso;

            if (recursoGdFl != null) {
                recursoGdFl.setScale(90);
                recursoGdFl.setAngle(30);
                recursoGdFl.setDither(false);
                recursoGdFl.setAlignWithLayer(true);
                recursoGdFl.setReverse(false);

                break;
            }
        }
        imagem.save(arquivoDestino, new PsdOptions());
    }

{{< /highlight >}}

    String arquivoFonte = "src/main/resources/input_1827.psd";
